WORK API 文档

host: http://api.work.qoder.cn

1.login

{
    path:/auth/login,
    method:POST,
    fileds:{
        username:String,
        password:String
    }
}

2.register

{
    path:/auth/register,
    method:POST,
    fileds:{
        username:String,
        password:String
    }
}

3.get UserInfo

{
    path:/user/info,
    method:GET,
    headers:{
        token:String
    }
}

4.get email list

{
    path:/email,
    method:GET,
    headers:{
        token:String
    }
}

5.add email

{
    path:/email,
    method:POST,
    headers:{
        token:String
    },
    fields:{
        belongTo:String,
        email:String,
        pass:String,
        stuCode:String
    }
}

6.update email

{
    path:/email,
    method:PUT,
    headers:{
        token:String
    },
    fields:{
        belongTo:String,
        email:String,
        pass:String,
        stuCode:String
    }
}

7.delete email

{
    path:/email/:id,
    method:DELETE,
    headers:{
        token:String
    }
}

8.get works

{
    path:/work,
    method:GET,
    headers:{
        token:String
    },
    queryString{
        email:String,
        times:String,
        belongTo:String
    }     //do not take unnecessary  query filed
}

9.add work

{
    path:/work,
    method:POST,
    headers:{
        token:String
    }，
    fields:{
        mailId:String,
        sendTo:String,
        times:String,
        lesson:String,
        content:String
    }
}
